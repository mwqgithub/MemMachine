set -Eeuo pipefail

# Config (override via env)
# ----------------------------
PG_VERSION=\"${PG_VERSION:-16}\"
POSTGRES_USER=\"${POSTGRES_USER:-memmachine}\"          # DB 超级用户（对应 docker 的 POSTGRES_USER）
POSTGRES_PASSWORD=\"${POSTGRES_PASSWORD:-mammachine_password}\" # DB 超级用户密码（对应 docker 的 POSTGRES_PASSWORD）
POSTGRES_DB=\"${POSTGRES_DB:-memmachine}\"              # 目标数据库（对应 docker 的 POSTGRES_DB）
POSTGRES_PORT=\"${POSTGRES_PORT:-5432}\"

POSTGRES_INITDB_ARGS=\"${POSTGRES_INITDB_ARGS:---encoding=UTF8 --lc-collate=C --lc-ctype=C}\"

# Windows 服务账户（本地用户“postgres”）的密码：必须满足 Windows 复杂性策略
PG_SERVICE_PASSWORD=\"${PG_SERVICE_PASSWORD:-P0stgresSvc!234}\"

# Paths (Windows format)
# ----------------------------
PG_ROOT=\"C:\\\\Program Files\\\\PostgreSQL\\\\${PG_VERSION}\"
PG_BIN=\"${PG_ROOT}\\\\bin\"
PG_DATA=\"${PG_ROOT}\\\\data\"
PG_SERVICE_NAME=\"postgresql-x64-${PG_VERSION}\"

# Helpers
ps() {
  powershell -NoLogo -NoProfile -NonInteractive -ExecutionPolicy Bypass -Command \"$@\"
}

# Convert MSYS path to Windows path
to_win() {
  cygpath -w \"$1\"
}

if [ ! -x \"/c/Program Files/PostgreSQL/${PG_VERSION}/bin/psql.exe\" ]; then
